  export default class MockData {
    static getInfos = async (id) => {
      const res = USER_MAIN_DATA.find((el) => el.id === parseInt(id));
      return { data: res };
    };
  
    static getActivity = async (id) => { 
      const res = USER_ACTIVITY.find((el) => el.userId === parseInt(id));
      return { data: res };
